同层比较策略

React 只比较同一层级的节点，不会跨层级移动。
相同位置、相同类型的节点才会复用，否则销毁重建。

function updateElement(fiber, element) {
  if (fiber?.type === element.type) {
    // ✅ 类型相同 → 复用
    return useFiber(fiber, element.props)
  }
  // ❌ 类型不同 → 销毁重建
  return createFiberFromElement(element)
}
